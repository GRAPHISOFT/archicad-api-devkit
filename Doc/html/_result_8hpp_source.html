655template<typename T, typename E>
656struct Storage {
657 static constexpr size_t Size = sizeof(T) > sizeof(E) ? sizeof(T) : sizeof(E);
658 static constexpr size_t Align = sizeof(T) > sizeof(E) ? alignof(T) : alignof(E);
659
660 typedef typename std::aligned_storage<Size, Align>::type type;
661
662 Storage()
663 : initialized_(false)
664 { }
665
666 void construct(types::Ok<T> ok)
667 {
668 new (&storage_) T(ok.val);
669 initialized_ = true;
670 }
671 void construct(types::Err<E> err)
672 {
673 new (&storage_) E(err.val);
674 initialized_ = true;
675 }
676
677 template<typename U>
678 void rawConstruct(U&& val) {
679 typedef typename std::decay<U>::type CleanU;
680
681 new (&storage_) CleanU(std::forward<U>(val));
682 initialized_ = true;
683 }
684
685 template<typename U>
686 const U& get() const {
687 return *reinterpret_cast<const U *>(&storage_);
688 }
689
690 template<typename U>
691 U& get() {
692 return *reinterpret_cast<U *>(&storage_);
693 }
694
695 template<typename U>
696 const U* getPtr() const {
697 return reinterpret_cast<const U *>(&storage_);
698 }
699
700 template<typename U>
701 U* getPtr() {
702 return reinterpret_cast<U *>(&storage_);
703 }
704
705 void destroy(ok_tag) {
706 if (initialized_) {
707 get<T>().~T();
708 initialized_ = false;
709 }
710 }
711
712 void destroy(err_tag) {
713 if (initialized_) {
714 get<E>().~E();
715 initialized_ = false;
716 }
717 }
718
719 type storage_;
720 bool initialized_;
721};

844template<typename T, typename E = Error>
845class Result : public ResultBase {
846public:
847 static_assert(!std::is_same<E, void>::value, "void error type is not allowed");
848
849 typedef details::Storage<T, E> storage_type;
850
851 Result(types::Ok<T> ok)
852 : ok_(true), token_(APINULLGuid)
853 {
854 storage_.construct(std::move(ok));
855 }
856
857 Result(types::Err<E> err, const API_Token& token)
858 : ok_(false), token_(token)
859 {
860 storage_.construct(std::move(err));
861 }
862
863 Result(Result&& other) {
864 if (other.IsOk()) {
865 details::Constructor<T, E>::move(std::move(other.storage_), storage_, details::ok_tag());
866 ok_ = true;
867 } else {
868 details::Constructor<T, E>::move(std::move(other.storage_), storage_, details::err_tag());
869 ok_ = false;
870 }
871 }
872
873 Result(const Result& other) {
874 if (other.IsOk()) {
875 details::Constructor<T, E>::copy(other.storage_, storage_, details::ok_tag());
876 ok_ = true;
877 } else {
878 details::Constructor<T, E>::copy(other.storage_, storage_, details::err_tag());
879 ok_ = false;
880 }
881 }
882
883 ~Result() {
884 if (ok_)
885 storage_.destroy(details::ok_tag());
886 else
887 storage_.destroy(details::err_tag());
888 }
889
891 operator bool() const {
892 return ok_;
893 }
894
896 bool IsOk() const {
897 return ok_;
898 }
899
901 bool IsErr() const {
902 return !ok_;
903 }
904
913 T Expect(std::string_view message) const {
914 if (!IsOk()) {
915 HandleUnwrapFailure (message, token_);
916 }
917 return expect_impl(std::is_same<T, void>());
918 }
