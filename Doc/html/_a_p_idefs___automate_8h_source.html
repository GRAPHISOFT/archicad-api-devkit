449struct API_IFCTranslatorIdentifier {
450
454 GSHandle innerReference;
455
459 GS::UniString name;
460
461 API_IFCTranslatorIdentifier ():
462 innerReference (nullptr),
463 name ("")
464 {
465 }
466
467 API_IFCTranslatorIdentifier (const API_IFCTranslatorIdentifier& that):
468 innerReference (nullptr),
469 name (that.name)
470 {
471 BMHandleToHandle (that.innerReference, &innerReference);
472 }
473
474 API_IFCTranslatorIdentifier (API_IFCTranslatorIdentifier&& that):
475 API_IFCTranslatorIdentifier ()
476 {
477 Swap (that);
478 }
479
480 ~API_IFCTranslatorIdentifier ()
481 {
482 BMKillHandle (&innerReference);
483 }
484
485 API_IFCTranslatorIdentifier& operator=(API_IFCTranslatorIdentifier other)
486 {
487 Swap (other);
488 return *this;
489 }
490
491 void Swap (API_IFCTranslatorIdentifier& other)
492 {
493 std::swap (innerReference, other.innerReference);
494 name.Swap (other.name);
495 }
496
497 friend void Swap (API_IFCTranslatorIdentifier& left, API_IFCTranslatorIdentifier& right)
498 {
499 left.Swap (right);
500 }
